Weathering Indices (WIs) computation

# Chemical Index of Weathering (CIW) after Harnois, 1988
CIW = signif(100 * molar_A / ( molar_A + molar_C + molar_N), digits = 2)                                                                        

# Chemical Proxy of Alteration (CPA) after Buggle et al., 2011  
CPA = signif(100 * molar_A / ( molar_A + molar_N), digits = 2)  

# Chemical Index of Alteration (CIA) after Nesbitt and Young, 1982
CIA = signif(100 * molar_A / ( molar_A + molar_C + molar_N + molar_K), digits = 2)                                                      

# Plagioclase Index of Alteration (PIA) after Fedo et al., 1995
PIA = signif(100 * (molar_A - molar_K)/ (molar_A + molar_C + molar_N - molar_K), digits = 2)                                            

# Modified Chemical Index of Alteration (CIX) after Garzanti et al., 2014
CIX = signif(100 * molar_A / ( molar_A + molar_N + molar_K), digits = 2)                                                                

# Index of Compositional Variability (ICV) after Cox et al., 1995
ICV = signif((molar_F + molar_K + molar_N + molar_C + molar_M + molar_Mn + molar_T) / molar_A, digits = 2)                              

# Weathering Index of Parker (WIP) after Parker, 1970
WIP = signif(100*(2*molar_N/0.35 + molar_M/0.9 + 2*molar_K/0.25 + molar_C/0.7), digits = 2)                                             

# Ruxton Ratio (R) after Ruxton, 1968
R = signif(molar_S/molar_A, digits = 2)                                                                                                 

# Vogts Residual Index (V) after Vogt, 1927, Roaldset, 1972
V = signif((molar_A + molar_K)/(molar_M + molar_C + molar_N), digits = 2)                                                               

# SilicaTitania Index (STI) after de Jayawardena and Izawa, 1994
STI = signif(100*(molar_S / molar_T)/(molar_S /molar_T + molar_S / molar_A + molar_A / molar_T), digits = 2)   

# Modified Weathering Potential Index (MWPI) after Reiche, 1943, Vogel, 1975
MWPI = signif(100 * (molar_N + molar_K + molar_C + molar_M) / ( molar_N + molar_K + molar_C + molar_M + molar_S + molar_A), digits = 2)

eight oxides (SiO2, TiO2, Al2O3, Fe2O3, MgO, CaO, Na2O and K2O) used in the formulas and re-closing to 100 wt.% (expressed as OXIDE.100 = 100*OXIDE/sum)

sum = elements$SiO2 + elements$TiO2 + elements$Al2O3 + elements$Fe2O3 + elements$MgO + elements$CaO + elements$Na2O + elements$K2O
SiO2.100 = 100*elements$SiO2/sum
TiO2.100 = 100*elements$TiO2/sum
Al2O3.100 = 100*elements$Al2O3/sum
Fe2O3.100 = 100*elements$Fe2O3/sum
MgO.100 = 100*elements$MgO/sum
CaO.100 = 100*elements$CaO/sum
Na2O.100 = 100*elements$Na2O/sum
K2O.100 = 100*elements$K2O/sum

M - F - W apices calculation after Ohta and Arai, 2007

# Step 1 of the formulas for calculating vertices (M, F and W) of M - F - W compositional space diagram
M = -0.395*log(SiO2.100) + 0.206*log(TiO2.100) - 0.316*log(Al2O3.100) + 0.160*log(Fe2O3.100) + 0.246*log(MgO.100) + 0.368*log(CaO.100) + 0.073*log(Na2O.100) - 0.342*log(K2O.100)+2.266
F = 0.191*log(SiO2.100) - 0.397*log(TiO2.100) + 0.020*log(Al2O3.100) - 0.375*log(Fe2O3.100) - 0.243*log(MgO.100) + 0.079*log(CaO.100) + 0.392*log(Na2O.100) + 0.333*log(K2O.100)-0.892
W = 0.203*log(SiO2.100) + 0.191*log(TiO2.100) + 0.296*log(Al2O3.100) + 0.215*log(Fe2O3.100) - 0.002*log(MgO.100) - 0.448*log(CaO.100) - 0.464*log(Na2O.100) + 0.008*log(K2O.100)-1.374

# Step 2 of the formulas for calculating vertices (M, F and W) of M - F - W compositional space diagram 
M = signif(exp(M)/(exp(M) + exp(F) + exp(W)), digits = 2)
F = signif(exp(F)/(exp(M) + exp(F) + exp(W)), digits = 2)
W = signif(exp(W)/(exp(M) + exp(F) + exp(W)), digits = 2)
